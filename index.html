// Collez ce code dans Apps Script
const SHEET='Journal';
function doGet(){
  const s=SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET);
  const r=s.getDataRange().getValues(); const h=r[0];
  return ContentService.createTextOutput(JSON.stringify(
    r.slice(1).map(row=>{const o={};h.forEach((k,i)=>o[k]=row[i]);return o;})
  )).setMimeType(ContentService.MimeType.JSON);
}
function doPost(e){
  const s=SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET);
  const p=JSON.parse(e.postData.contents);
  if(p.action==='sync'){
    const last=s.getLastRow(); if(last>1) s.deleteRows(2,last-1);
    const rows=p.data.map(r=>[r.numOp,r.date,r.moisOp,r.designation,r.classe,r.rubrique,r.par,r.entree,r.sortie,r.tel,r.commentaire,r.matricule]);
    if(rows.length) s.getRange(2,1,rows.length,12).setValues(rows);
  }
  return ContentService.createTextOutput('OK');
}
function init(){
  const s=SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET)||SpreadsheetApp.getActiveSpreadsheet().insertSheet(SHEET);
  s.getRange(1,1,1,12).setValues([['numOp','date','moisOp','designation','classe','rubrique','par','entree','sortie','tel','commentaire','matricule']]);
}
